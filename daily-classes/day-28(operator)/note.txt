// ======================================================
// ðŸ”¸ ARITHMETIC OPERATORS (+, -, *, /, %, **)
// ======================================================

1 + 2           // 3
'ha' + 'rsh'    // 'harsh'  â†’ concatenation
12 - 2          // 10
12 / 6          // 2

// ======================================================
// ðŸ”¸ COMPARISON OPERATORS (==, ===, !=, !==, <, >, <=, >=)
// ======================================================

// "=" â†’ Assignment (used to give value)
let a = 12; // assignment operator

12 == 13       // false
12 == '12'     // true  â†’ checks only value (not type)
12 === '12'    // false â†’ checks value + type (strict)
12 != 13       // true
12 != '12'     // false â†’ checks only value
12 !== '12'    // true  â†’ checks value + type

12 > 15        // false
12 < 15        // true
22 < 22        // false
22 <= 22       // true

// Summary:
// == and != â†’ loose comparison (only value)
// === and !== â†’ strict comparison (value + type)


// ======================================================
// ðŸ”¸ ASSIGNMENT OPERATORS (=, +=, -=, *=, /=, %=)
// ======================================================
let x = 12;
x += 3;   // x = 15
x -= 11;  // x = 4
x *= 2;   // x = 8
x /= 2;   // x = 4
x %= 3;   // x = 1


// ======================================================
// ðŸ”¸ LOGICAL OPERATORS (&&, ||, !)
// ======================================================
//
// && â†’ AND â†’ both sides must be true
// || â†’ OR  â†’ at least one side true
// !  â†’ NOT â†’ reverses truthiness

true && true      // true
false && true     // false
12 > 13 && 13 > 10 // false

12 > 13 || 13 > 10 // true
false || true      // true

!false             // true
!12                // false (12 â†’ truthy â†’ negated â†’ false)
!!12               // true  (double negation â†’ truthy)


// ======================================================
// ðŸ”¸ UNARY OPERATORS (+, -, !, typeof, ++, --)
// ======================================================

+'15'          // 15
+'harsh'       // NaN
-5             // -5
!12            // false

typeof 12           // 'number'
typeof '12'         // 'string'
typeof []           // 'object'
typeof function(){} // 'function'

let a1 = 12;
++a1;               // 13  (pre-increment)
a1++;               // 13 (shows 13, then becomes 14)

let b1 = 15;
b1++;               // 15 (now 16)
++b1;               // 17

let c1 = 2;
c1++ + c1;          // 2 + 3 = 5

let d1 = 3;
--d1;               // 2
d1--;               // 2 (now becomes 1)


// ======================================================
// ðŸ”¸ TERNARY OPERATOR ( ? : )
// ======================================================
// Syntax: condition ? trueExpression : falseExpression

12 > 13 ? console.log(true) : console.log(false);
// Output: false


// ======================================================
// ðŸ”¸ typeof vs instanceof
// ======================================================

typeof 12               // 'number'
typeof null             // 'object'
typeof []               // 'object'
typeof {}               // 'object'
typeof function(){}     // 'function'
typeof NaN              // 'number'

// âŒ typeof [] and typeof null â†’ 'object' (incorrect categorization)

let arr = [];
arr instanceof Array    // true
let obj = {};
obj instanceof Object   // true
obj instanceof Array    // false

// instanceof â†’ used for reference types ([], {}, functions)
// typeof â†’ used for primitive types (number, string, boolean, etc.)

// ------------------------------------------------------

  case 1:

    console.log("Matched");

    console.log("Default");
}

  else return "A";

// ------------------------------------------------------
// ðŸ”¸ Q: Write function getGrade(score)
// ------------------------------------------------------
function getGrade(score) {
  if (score >= 90 && score <= 100) return "A+";
  if (score >= 80 && score <= 89) return "A";
  if (score >= 70 && score <= 79) return "B";
  if (score >= 60 && score <= 69) return "C";
  if (score >= 33 && score <= 59) return "D";
  if (score >= 0 && score <= 32) return "Fail";
  return "Invalid marks X";
}
console.log(getGrade(54)); // D


// ------------------------------------------------------
// ðŸ”¸ Rockâ€“Paperâ€“Scissors Logic
// ------------------------------------------------------

function rps(user, computer) {
  if (user === computer) return "Draw";
  if (user === "rock" && computer === "scissor") return "User";
  if (user === "scissor" && computer === "paper") return "User";
  if (user === "paper" && computer === "rock") return "User";
  return "Computer";
}
rps("rock", "rock"); // "Draw"

// ðŸ”¸ FOR LOOP
// ------------------------------------------------------

// ------------------------------------------------------

// ------------------------------------------------------
